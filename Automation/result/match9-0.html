class parp extends RecursiveAction {

	private static final long serialVersionUID = 2892201041359481917L;
	long n;
	long k;
	long res;

	public long getres() {
		return res;
	}

	public parp(long n, long k)
	{
		this.n=n;
		this.k=k;
	}

	@Override
	protected void compute() {
		if (n == 0) {
			this.res = 1;
			return;
		}
		else if(k==0){
			this.res = 1;
			return;
		}
		else if(n==k){
			this.res = 1;
			return;
		}

		parp right = new parp(n - 1, k);
		parp left = new parp(n - 1, k - 1);
		right.fork();
		left.compute();
		right.join();
		this.res = left.getres() + right.getres();
	}

}
